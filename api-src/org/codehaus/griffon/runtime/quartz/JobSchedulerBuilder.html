023 /**

028

031      */

035      * The {@code Job} name, must be unique within the group.

047      * encountered.
048      */
049     private boolean requestRecovery = false;

076     private TimeZone timeZone = getDefault();

078     /**

081      * first.

095      *
096      * @since 1.3

217     /**
218      * Sets the {@code Trigger} that will be used to schedule
219      * the {@code Job}.
220      * <p/>
221      * <p/>
222      * Be aware that using using this method will override any other
223      * {@code Trigger}-related operation, like {@link #withTriggerGroup(String)}
224      * or {@link #withTimeZone(TimeZone)}
225      *
226      * @param trigger The {@code Trigger} to associate with the {@code Job}
227      * @return This builder instance
228      * @since 1.2
229      */
230     public JobSchedulerBuilder withTrigger(Trigger trigger) {
231         this.trigger = trigger;
232         return this;
233     }

235     /**
236      * Requests an existing trigger (sharing the same key as the new trigger) for this job to
237      * be replaced with the new trigger.
238      *
239      * @return This builder instance
240      * @since 1.3
241      */
242     public JobSchedulerBuilder updateExistingTrigger() {
243         this.updateExistingTrigger = true;
244         return this;
245     }

247     /**
248      * Add the produced {@code Job} to the given {@code Scheduler},
249      * and associate the related {@code Trigger} with it.
250      * <p/>
251      *
252      * @param scheduler The given {@code Scheduler}
253      * @throws SchedulerException If any error occurs
254      */
255     public void schedule(@Nonnull Scheduler scheduler) throws SchedulerException {
256         if (cronExpression == null && trigger == null) {
257             throw new IllegalStateException(format("Impossible to schedule Job '%s' without cron expression",
258                 jobClass.getName()));
259         }
260         if (cronExpression != null && trigger != null) {
261             throw new IllegalStateException(format("Impossible to schedule Job '%s' with cron expression " +
262                 "and an associated Trigger at the same time", jobClass.getName()));
263         }
264 
265         JobKey jobKey = jobKey(DEFAULT.equals(jobName) ? jobClass.getName() : jobName, jobGroup);
266         TriggerKey triggerKey = triggerKey(DEFAULT.equals(triggerName) ? jobClass.getCanonicalName() : triggerName, triggerGroup);
267 
268         if (updateExistingTrigger && scheduler.checkExists(triggerKey)) {
269             scheduler.unscheduleJob(triggerKey);
270         }
271 
272         scheduler.scheduleJob(newJob(jobClass)
273                 .withIdentity(jobKey)
274                 .requestRecovery(requestRecovery)
275                 .storeDurably(storeDurably).build(),
276             (trigger == null) ?
277                 newTrigger()
278                     .withIdentity(triggerKey)
279                     .withSchedule(cronSchedule(cronExpression)
280                         .inTimeZone(timeZone))
281                     .withPriority(priority)
282                     .build()
283                 : trigger);
284     }
